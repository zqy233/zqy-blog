 // 周五
this.startDate = dayjs().startOf("week").add(5, "day").format("YYYY-MM-DD")
// 下周四
this.endDate = dayjs().startOf("week").add(11, "day").format("YYYY-MM-DD")
this.getMainNetOverhual()
