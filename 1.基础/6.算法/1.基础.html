дәҢеҲҶжҹҘжүҫзҡ„еҹәжң¬жҖқжғіжҳҜе°ҶnдёӘе…ғзҙ еҲҶжҲҗеӨ§иҮҙзӣёзӯүзҡ„дёӨйғЁеҲҶпјҢеҸ–arr[n/2]дёӯй—ҙеҖјдёҺxеҒҡжҜ”иҫғпјҢеҰӮжһңx=arr[n/2]пјҢеҲҷжүҫеҲ°xпјҢз®—жі•з»“жқҹпјӣеҰӮжһңx<arr[n/2]пјҢеҲҷеҸӘиҰҒеңЁж•°з»„arrзҡ„е·ҰеҚҠйғЁеҲҶз»§з»ӯжҹҘжүҫxпјӣеҰӮжһңx>arr[n/2]пјҢеҲҷеҸӘиҰҒеңЁж•°з»„arrзҡ„еҸіеҚҠйғЁеҲҶжҹҘжүҫx

arrжҳҜжүҖиҰҒжҹҘжүҫзҡ„ж•°з»„пјҢxжҳҜиҰҒжҹҘжүҫзҡ„е…ғзҙ 

js
/**
 * дәҢеҲҶжҹҘжүҫ
 * @param arr жҹҘжүҫзҡ„ж•°з»„
 * @param x жҹҘжүҫзҡ„е…ғзҙ 
 * @returns жҹҘжүҫеҲ°иҝ”еӣһж•°з»„дёӢж ҮпјҢеҗҰеҲҷиҝ”еӣһ-1
 */
function binarySearch(arr, x) {
  let low = 0 // йҰ–дёӢж Ү
  let high = arr.length - 1 // е°ҫдёӢж Ү
  // еҸӘиҰҒжҹҘжүҫеҢәй—ҙиө·е§ӢзӮ№е’Ңз»“жқҹзӮ№дёӯй—ҙиҝҳжңүеҖј(иҰҒеҢ…жӢ¬дёӨеҖјзӣёеҗҢзҡ„жғ…еҶө)пјҢе°ұз»§з»ӯиҝӣиЎҢжҹҘжүҫ
  while (low <= high) {
    const mid = (low + high) / 2 // зЎ®е®ҡдёӯй—ҙеҖјдёӢж Ү
    //еҰӮжһңжҹҘжүҫеҖјзӯүдәҺдёӯй—ҙеҖј
    if (x == arr[mid]) {
      return mid // еҲҷиҝҷдёӘmidеҖјпјҢе°ұжҳҜжҹҘжүҫеҲ°зҡ„ж•°з»„дёӢж Ү
    }
    //еҰӮжһңжҹҘжүҫеҖје°ҸдәҺдёӯй—ҙеҖј
    else if (x < arr[mid]) {
      high = mid - 1 // еҲҷеңЁе·ҰеҚҠйғЁеҲҶжҹҘжүҫпјҢйңҖиҰҒйҮҚж–°зЎ®и®ӨеҢәй—ҙhighзҡ„дҪҚзҪ®
    }
    // еҗҰеҲҷжҹҘжүҫеҖјеӨ§дәҺдёӯй—ҙеҖј
    else {
      low = mid + 1 // еҲҷеңЁеҸіеҚҠйғЁеҲҶжҹҘжүҫпјҢйңҖиҰҒйҮҚж–°зЎ®и®ӨеҢәй—ҙlowзҡ„дҪҚзҪ®
    }
  }
  return -1 //жІЎжңүжҹҘжүҫеҲ°пјҢиҝ”еӣһ-1
}
const arr = [1, 3, 5, 6, 7, 8, 9]
const x = 7
console.log(binarySearch(arr, x))
