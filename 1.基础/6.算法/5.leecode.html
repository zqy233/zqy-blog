function reverse(x: number): number {
	if(x == 0) return x
	var flag = x > 0 ? true : false
	var num:number =Number((Math.abs(x)+'').split('').reverse().join(''))
    var limit = Math.pow(2,31)
    if(!flag && num > limit){
		return 0
    }
    if(flag && num > (limit - 1)){
        return 0
    }
	return flag?num:-num
}
